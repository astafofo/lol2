import numpy as np
from tensorflow.keras.models import Sequential
from tensorflow.keras.layers import Conv1D, Flatten, Dense

X = np.random.rand(2,50,1)  # 50 bytes per packet
y = [0,1]
model = Sequential([Conv1D(4,3,activation='relu',input_shape=(50,1)),Flatten(),Dense(1,activation='sigmoid')])
model.compile('adam','binary_crossentropy'); model.fit(X,y,epochs=2,verbose=0)
print("Task15: Raw Packet Classification Done")
