import numpy as np
from tensorflow.keras.models import Sequential
from tensorflow.keras.layers import LSTM, Dense, Embedding

X = np.random.randint(1,10,(4,5))
y = np.array([0,1,0,1])  # 0=weak,1=strong
model = Sequential([Embedding(10,4,input_length=5), LSTM(8), Dense(1,activation='sigmoid')])
model.compile('adam','binary_crossentropy'); model.fit(X,y,epochs=2,verbose=0)
print("Task6: Password Prediction", (model.predict(X)>0.5).astype(int).flatten())
